library(recipes)

train %>% 
  recipe(arr_delay~.) %>% 
  step_rm(dep_delay, year, time_hour) %>% 
  step_center(all_numeric()) %>% 
  step_scale(all_numeric()) %>% 
  step_nzv(all_nominal()) %>% 
  prep() ->
  fe_cleaner
